import os import re from datetime import datetime BLOG_DIR = "landing/blog" def extract_title(html: str, fallback: str) -> str: match = re.search(r"(.*?)", html, re.S) if match: return match.group(1).strip() return fallback.replace(".html", "") def generate_blog_index(): os.makedirs(BLOG_DIR, exist_ok=True) files = [ f for f in os.listdir(BLOG_DIR) if f.endswith(".html") and f != "index.html" ] items = "" for f in sorted(files): path = os.path.join(BLOG_DIR, f) try: with open(path, "r", encoding="utf-8") as file: html = file.read() title = extract_title(html, f) except Exception: title = f.replace(".html", "") items += f'
  • {title}
    • \n' now = datetime.now().strftime("%Y-%m-%d") index_html = f""" ブログ一覧 | MCAddon Translator

    ブログ一覧

    更新日: {now}

      {items}

    MCAddon Translator トップへ

    """ with open(os.path.join(BLOG_DIR, "index.html"), "w", encoding="utf-8") as f: f.write(index_html) print("✅ blog index 生成完了")
